	l <= tail(l)

	return true

var size(list l)
int s <= 0
while empty(l) = false
	s <= s + 1
	l <= tail(l)
return s
	
// Fügt zwei listen zusammen

for var a <= size(x) to a >= 0, a <= a - 1
	cons(get(x, a), y)
return y

// Ergibt das kartesische Produkt r x s

\section*{Aufgabe 3}
\begin{enumerate}
 \item Für ein beliebiges Tupel $a_{n}$ mit Attributen $A, B, C$ gilt: Sollte es im Fall $x$ zuerst durch eine Projektion auf die Spalten A und B reduziert werden, wird danach jedes Element, für das $A>10$ gilt, zurückgeliefert. Bei $y$ werden im ersten Schritt exakt die selben Tupel ausgewählt, nämlich nur die, für welche auch $A>10$ gilt, und davon werden dann die zwei Attribute A und B zurückgegeben. Die Beiden Operationen Projektion und Auswahl (\texttt{select}) können in diesem Fall ausgetauscht werden, die beiden Ergebnisse sind äquivalent.
 \item Hier sind $x$ und $y$ nicht gleichbedeutend. Wenn ein Tupel $a_{1}$ einen Wert in r.A mit einem anderen Tupel $a_{2}$ in t.A gemeinsam hat, abet r.B (in $a_{1}$) nicht mit t.B (in $a_{2}$) übereinstimmt, wird $a_{1}$ in der Auswertung $\pi_{A}(r - t)$ erscheinen, nicht jedoch in $\pi_{A}(r) - \pi_{A}(t)$. Die Differenz wird hier nicht aus den selben Mengen gebildet.
\end{enumerate}
